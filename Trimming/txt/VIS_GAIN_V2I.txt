from dfttools import *
from time import sleep
from Procedures.Startup import startup
from Procedures.Global_enable import global_enable
from Procedures.VI_SNS_turn_on import vi_sns_turn_on
from Procedures.VI_SNS_turn_off import vi_sns_turn_off

from time import sleep
def vis_gain_v2i():
  Testname = 'VIS_GAIN_V2I'
  print(f' {Testname}')

  '''
  Due to technological mismatch of input and DAC resistors, when a voltage at the input is applied, a current is flowing to those resistors. This effect produces a current modifying the signal current that must be measured. To compensate for this effect, the procedure is the following:
  1. Remove output load from class-D output
  2. Force class-D output to have SPKRP high and SPKRN low.
  3. Read I-sense output digital code.
  4. Force class-D output to have SPKRP low and SPKRN high.
  5. Read I-sense output digital code
  GVI = DELTA_V=(2*VDDP)/DELTA_code

  '''

  print(f' ........ Remove load from classD ........')
  startup()
  global_enable()
  vi_sns_turn_on()

  sleep(0.001)
  [i2c_write] device_addr=0x38, PageName=PAG0, register_addr=0xFE, data=0x0, Mask=0x1, Default=0x0
  [i2c_write] device_addr=0x38, PageName=PAG0, register_addr=0x9C, data=0x0, Mask=0xFF, Default=0x0
  [i2c_write] device_addr=0x38, PageName=PAG0, register_addr=0x9B, data=0x10, Mask=0x10, Default=0x81
  [i2c_write] device_addr=0x38, PageName=PAG0, register_addr=0x9C, data=0x66, Mask=0xFF, Default=0x0  # set bridge switch (LSp off, MIDp off, CASp off, HSp off, LSn off, MIDn off, CASn off, HSn off)
  # read the isense value before trimming 
  i_sense_code_before_trim = [i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x6B, Mask=0xFF
[i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x6C, Mask=0xFF
  print(f'i_sense_code_before_trim : {hex(i_sense_code_before_trim)}')
  sleep(0.1)
  # Put the Bridege in High Impedence 
  [i2c_write] device_addr=0x38, PageName=PAG0, register_addr=0x9C, data=0x0, Mask=0xFF, Default=0x0
  VFORCE("OUTP","OUTN",2.5,0)
  sleep(1)
  vsns_gain_high_measured = [i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x69, Mask=0xFF
[i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x6A, Mask=0xFF
  vsns_gain_high_measured =  0x10000 - vsns_gain_high_measured if vsns_gain_high_measured & 0x8000 else vsns_gain_high_measured
  print(f'vsns_gain_high_measured : {hex(vsns_gain_high_measured)}')
  isns_gain_high_measured = [i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x6B, Mask=0xFF
[i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x6C, Mask=0xFF
  isns_gain_high_measured =   0x10000 - isns_gain_high_measured    if isns_gain_high_measured & 0x8000 else isns_gain_high_measured
  print(f'isns_gain_high_measured : {hex(isns_gain_high_measured)}')
  VFORCE ("OUTP","OUTN",-2.5,0)
  sleep(1)
  vsns_gain_low_measured = [i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x69, Mask=0xFF
[i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x6A, Mask=0xFF
  vsns_gain_low_measured = 0x10000 - vsns_gain_low_measured   if vsns_gain_low_measured & 0x8000 else vsns_gain_low_measured
  print(f'vsns_gain_low_measured : {hex(vsns_gain_low_measured)}')
  isns_gain_low_measured = [i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x6B, Mask=0xFF
[i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x6C, Mask=0xFF
  isns_gain_low_measured =   0x10000 - isns_gain_low_measured  if isns_gain_low_measured & 0x8000 else isns_gain_low_measured
  print(f'isns_gain_low_measured : {hex(isns_gain_low_measured)}')
  
  LSB_expander = 64/2**20 # expand to know how many lsb's
  v2i_gain = int(((vsns_gain_high_measured+vsns_gain_low_measured) / (isns_gain_high_measured - isns_gain_low_measured) )  ) & 0xFFF # convert it to the 12bit value
  # change the sign for the gain code 
  v2i_gain_code = 0x1000 + v2i_gain  if (v2i_gain & 0x800) else 0x1000 - v2i_gain
  # write the v2i_gain_code 
  print(f'v2i_gain_code : {hex(v2i_gain_code)}')
  [i2c_write] device_addr=0x38, PageName=PAG1, register_addr=0xFE, data=0x1, Mask=0x1, Default=0x0
  [i2c_write] device_addr=0x38, PageName=PAG1, register_addr=0xBF, data=v2i_gain_code, Mask=0xF, Default=0x0
[i2c_write] device_addr=0x38, PageName=PAG1, register_addr=0xC0, data=v2i_gain_code, Mask=0xFF, Default=0x0
  # read corrected isense value 
  [i2c_write] device_addr=0x38, PageName=PAG0, register_addr=0xFE, data=0x0, Mask=0x1, Default=0x0
  i_sense_code_after_trim = [i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x6B, Mask=0xFF
[i2c_read] device_addr=0x38, PageName=PAG0, register_addr=0x6C, Mask=0xFF
  print(f'i_sense_code_after_trim : {hex(i_sense_code_after_trim)}')
  # turn off the power supply on Outp Vs Outn
  VFORCE ("OUTP","OUTN",float('Inf'),0)
  # turn off the driver 
  [i2c_write] device_addr=0x38, PageName=PAG0, register_addr=0x9B, data=0x0, Mask=0x10, Default=0x81
  # turn off vi sense block 
  vi_sns_turn_off()
if __name__ == '__main__':
  vis_gain_v2i()
